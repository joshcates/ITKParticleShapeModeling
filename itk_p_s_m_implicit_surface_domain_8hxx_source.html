28 template<class T, unsigned int VDimension>
29 void
30 PSMImplicitSurfaceDomain<T, VDimension>::
31 SetCuttingPlane(const vnl_vector<double> &a, const vnl_vector<double> &b,
32  const vnl_vector<double> &c)
33 {
34  // See http://mathworld.wolfram.com/Plane.html, for example
35  vnl_vector<double> q;
36  if (VDimension == 3) q = vnl_cross_3d((b-a),(c-a));
37  else if (VDimension == 2) q = vnl_cross_2d((b-a),(c-a));
38 
39  if (q.magnitude() > 0.0)
40  {
41  m_CuttingPlaneNormal = q / q.magnitude();
42  m_CuttingPlanePoint = a;
43  m_UseCuttingPlane = true;
44  }
45 }

47 template<class T, unsigned int VDimension>
48 bool
49 PSMImplicitSurfaceDomain<T, VDimension>::
50 ApplyVectorConstraints(vnl_vector_fixed<double, VDimension> &gradE,
51  const PointType &pos,
52  double maxtimestep) const
53 {
54  if (this->m_UseCuttingPlane == true)
55  {
56  // See http://mathworld.wolfram.com/Point-PlaneDistance.html, for example
57  vnl_vector_fixed<double, 3> x;
58  vnl_vector_fixed<T, VDimension> grad = this->SampleGradientVnl(pos);
59  for (unsigned int i = 0; i < VDimension; i++)
60  { x[i] = pos[i]; }
61  const double D = dot_product(m_CuttingPlaneNormal, x- m_CuttingPlanePoint);
62 
63  // x = m_CuttingPlaneNormal * fabs(1.0 / (D + 1.0e-3));
64  // x = m_CuttingPlaneNormal * lambda * exp(-lambda * fabs(D));
65 
66  // Gradient of simple force 1/D^2 to push away from cutting plane
67  vnl_vector_fixed<double, VDimension> df;
68  const double k = (-2.0 / (D * D * D));
69  df[0] = k * grad[0] * m_CuttingPlaneNormal[0];
70  df[1] = k * grad[1] * m_CuttingPlaneNormal[1];
71  df[2] = k * grad[2] * m_CuttingPlaneNormal[2];
72 
73  gradE = gradE + df;
74 
75  // Make sure force is not huge relative to other forces.
76  if (gradE.magnitude() > maxtimestep)
77  {
78  gradE.normalize();
79  gradE = gradE * maxtimestep;
80  }
81  }
82 
83  return Superclass::ApplyVectorConstraints(gradE,pos,maxtimestep);
84 }

86 template<class T, unsigned int VDimension>
87 bool
88 PSMImplicitSurfaceDomain<T, VDimension>::ApplyConstraints(PointType &p) const
89 {
90  // First apply and constraints imposed by superclasses. This will
91  // guarantee the point starts in the correct image domain.
92  bool flag = Superclass::ApplyConstraints(p);
93 
94  if (this->m_ConstraintsEnabled == true)
95  {
96 
97  unsigned int k = 0;
98  double mult = 1.0;
99 
100  const T epsilon = m_Tolerance * 0.001;
101  T f = this->Sample(p);
102 
103  T gradmag = 1.0;
104  while ( fabs(f) > (m_Tolerance * mult) || gradmag < epsilon)
105  // while ( fabs(f) > m_Tolerance || gradmag < epsilon)
106  {
107  vnl_vector_fixed<T, VDimension> grad = this->SampleGradientVnl(p);
108 
109  gradmag = grad.magnitude();
110  vnl_vector_fixed<T, VDimension> vec = grad * ( f / (gradmag + epsilon) );
111  for (unsigned int i = 0; i < VDimension; i++)
112  {
113  p[i] -= vec[i];
114  }
115 
116  f = this->Sample(p);
117 
118  // Raise the tolerance if we have done too many iterations.
119  k++;
120  if (k > 10000)
121  {
122  mult *= 2.0;
123  k = 0;
124  }
125  } // end while
126  } // end if m_ConstraintsEnabled == true
127 
128  return flag;
129 }

131 template <class T, unsigned int VDimension>
132 double
133 PSMImplicitSurfaceDomain<T, VDimension>::Distance(const PointType &a, const PointType &b) const
134 {
135  double sum = 0.0;
136  for (unsigned int i = 0; i < VDimension; i++)
137  { sum += (b[i]-a[i]) * (b[i]-a[i]); }
138  return sqrt(sum);
139 }
